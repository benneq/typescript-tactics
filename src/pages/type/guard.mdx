---
title: Type Guard
tags: ['type', 'guard']
---

### User-Defined

```ts
const isString = (value: unknown): value is string => {
  return typeof value === 'string';
};

if (isString(value)) {
  type Value = string;
}
```

### Inline

```ts
const result = input.filter((value): value is boolean => value !== null);
type Result = boolean[];
```

### `in` opterator

```ts
if ('prop' in value) {
  value.prop;
}
```

### `typeof`

```ts
if (typeof value === 'string') {
  type Value = string;
}
```

### `instanceof`

```ts
if (value instanceof Date) {
  type Value = Date;
}
```
